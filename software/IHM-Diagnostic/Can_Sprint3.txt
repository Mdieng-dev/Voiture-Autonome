void donnee_oil (void const* arg){

	}

}

// La fonction de callback (Interruption)
void CAN_callback(uint32_t obj_idx, uint32_t event) {
	  
		char *ptr_US;
	  char *ptr_GPS;
		ptr_US = (char *) &datarecu_US;
	  ptr_GPS = (char *) &datarecu_GPS;
    if (event & ARM_CAN_EVENT_RECEIVE) {	
        Driver_CAN1.MessageRead(0, &rx_msg_info, data_buf, 8);
			  ID_donnees = rx_msg_info.id;
        switch(rx_msg_info.id){
					case 0x5D7 : // test 
							osThreadFlagsSet (ID_test, 0x01);
						break;			
					case 0x110 : // ultrasons avant
						*ptr_US = data_buf[0];
						*(ptr_US+1) = data_buf[1];
						*(ptr_US+2) = data_buf[2];
						*(ptr_US+3) = data_buf[3];
						
						osThreadFlagsSet (ID_US, 0x01);
						break;
					case 0x111 : // ultrasons avant
						*ptr_US = data_buf[0];
						*(ptr_US+1) = data_buf[1];
						*(ptr_US+2) = data_buf[2];
						*(ptr_US+3) = data_buf[3];
						
						osThreadFlagsSet (ID_US_ar, 0x01);
						break;
					case 0x131: // Phares
						break;
					case 0x140 : // RFID
							osThreadFlagsSet (ID_rfid, 0x01);
						break;
					case 0x170 : // gps
						*ptr_US = data_buf[0];
						*(ptr_GPS+1) = data_buf[1];
						*(ptr_GPS+2) = data_buf[2];
						*(ptr_GPS+3) = data_buf[3];
					  *(ptr_GPS+4) = data_buf[4];
					  *(ptr_GPS+5) = data_buf[5];
					  *(ptr_GPS+6) = data_buf[6];
					  *(ptr_GPS+7) = data_buf[7];
					
							osThreadFlagsSet (ID_gps, 0x01);
						break;
					case 0x132: // 0x200 vitesse (normalement) from 0x0f6 to 0x132 puisqu'on a mixť vitesse et clignotant
						osThreadFlagsSet (ID_vitesse, 0x01);
						break;
					case 0x250 : // capteur liquide
							osThreadFlagsSet (ID_oil, 0x01);
						break;

					default: break;
				}
     }
	 }

void Initialisation_CAN(void) {
    Driver_CAN1.Initialize(NULL, CAN_callback);
    Driver_CAN1.PowerControl(ARM_POWER_FULL);
    Driver_CAN1.SetMode(ARM_CAN_MODE_INITIALIZATION);
    Driver_CAN1.SetBitrate(ARM_CAN_BITRATE_NOMINAL, 
													125000, 
                          ARM_CAN_BIT_PROP_SEG(3U) | 
													ARM_CAN_BIT_PHASE_SEG1(1U) | 
                          ARM_CAN_BIT_PHASE_SEG2(1U) | 
													ARM_CAN_BIT_SJW(1U));
    
    // Configurer le filtre et l'objet de rťception
    Driver_CAN1.ObjectConfigure(0, ARM_CAN_OBJ_RX); 
	  Driver_CAN1.ObjectSetFilter( 0, ARM_CAN_FILTER_ID_MASKABLE_ADD , ARM_CAN_STANDARD_ID(0x170),0x000);
	
    Driver_CAN1.SetMode(ARM_CAN_MODE_NORMAL);
}
