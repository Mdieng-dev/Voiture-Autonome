/*********************************************************************
*
*       SCREEN_Energie__WM_NOTIFICATION_VALUE_CHANGED__ID_TEXT_03__APPW_JOB_SETVALUE
*/
void SCREEN_Energie__WM_NOTIFICATION_VALUE_CHANGED__ID_TEXT_03__APPW_JOB_SETVALUE(APPW_ACTION_ITEM * pAction, WM_HWIN hScreen, WM_MESSAGE * pMsg, int * pResult) {
	  APPW_SetValue(SCREEN_Conduite,ID_GAUGE_00,speed);
		switch(ID_donnees){
					case 0x170 : // GPS
							APPW_SetValue(SCREEN_Conduite,ID_TEXT_07,longi);
							APPW_SetValue(SCREEN_Conduite,ID_TEXT_05,lat);
						break;			
					case 0x132 : //VITESSE & CLIGNOTANT
						if(clignotant == 1){
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_gauche,1);
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_droit,0);
						}
						else if(clignotant == 2){
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_gauche,0);
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_droit,1);
						}
						else{
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_gauche,0);
							APPW_SetValue(SCREEN_Conduite,ID_SWITCH_droit,0);
						}
						break;
					default:
						break;
				}
}
